/* DOCUMENTATION: Connected to a live backend endpoint using asynchronous fetch requests */ async function sendToAPI(promptText) { // 1. Save the user's message to our local array tracking the conversation state convoHistory.push({ role: "user", content: promptText }); try { // 2. Make a real network request to our secure backend server const response = await fetch('/api/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, // Pass the entire history so the AI remembers past details body: JSON.stringify({ history: convoHistory, path: selectedPath }) }); if (!response.ok) throw new Error('Network response failed'); const data = await response.json(); // 3. Save the AI's reply to our conversation history tracking convoHistory.push({ role: "assistant", content: data.reply }); // 4. Clean up the typing dots and print out the AI's response text resolveTyping(data.reply); // 5. Check the instructions sent back by the server to decide what UI to show next if (data.action === "show_ideas") { setTimeout(() => { // Dynamically build tiles based on the custom JSON options returned by the live AI renderLiveIdeas(data.ideas); }, 1500); } else { // The AI wants to continue clarifying, append a fresh user reply text area appendReplyInput(); } } catch (error) { console.error("API Error:", error); resolveTyping("Sorry, I encountered an issue connecting to the engine. Please try again."); appendReplyInput(); } } /* DOCUMENTATION: Helper to build the visual tiles dynamically from real AI outputs */ function renderLiveIdeas(ideasArray) { const convo = document.getElementById('convo-thread'); if(convo) convo.style.display = 'none'; const panel = document.getElementById('ideas-panel'); if(panel) panel.style.display = 'block'; window.scrollTo({top: 0, behavior: 'smooth'}); const grid = document.getElementById('ideas-grid'); if(!grid) return; // Clear out any old mock templates grid.innerHTML = ''; // Map through the array generated directly by the live language model ideasArray.forEach((idea, index) => { const tile = document.createElement('div'); tile.className = 'idea-tile'; tile.onclick = () => viewJourney(idea.name); tile.innerHTML = `
Option ${index + 1} • Tailored Concept
${esc(idea.name)}
${esc(idea.tagline)}
${fmt(idea.whyFits)}
`; grid.appendChild(tile); }); }
